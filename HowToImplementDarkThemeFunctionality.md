# How to Implement Dark Theme Functionality

This guide explains how to implement a complete dark theme toggle functionality in a Next.js project using Tailwind CSS v4, React Context, and TypeScript.

## Prerequisites

- Next.js project with Tailwind CSS v4
- React 19+
- TypeScript (optional but recommended)

## Step 1: Configure Tailwind CSS v4 for Class-Based Dark Mode

In your `app/globals.css` file, add the `@custom-variant` directive to enable class-based dark mode:

```css
@import "tailwindcss";

@custom-variant dark (&:where(.dark, .dark *));

:root {
  --background: #ffffff;
  --foreground: #171717;
}

:root.dark {
  --background: #0a0a0a;
  --foreground: #ededed;
}

body {
  background: var(--background);
  color: var(--foreground);
  font-family: var(--font-geist-sans), Arial, Helvetica, sans-serif;
}
```

**Important:** The `@custom-variant dark (&:where(.dark, .dark *));` line is crucial for Tailwind CSS v4. It tells Tailwind to apply `dark:` prefixed classes when the `.dark` class is present on any parent element.

## Step 2: Create the ThemeProvider Component

Create a new file `components/ThemeProvider.tsx`:

```tsx
"use client";

import { createContext, useContext, useEffect, useState } from "react";

type Theme = "light" | "dark";

type ThemeContextType = {
  theme: Theme;
  toggleTheme: () => void;
  mounted: boolean;
};

const ThemeContext = createContext<ThemeContextType | undefined>(undefined);

export function ThemeProvider({ children }: { children: React.ReactNode }) {
  const [theme, setTheme] = useState<Theme>("light");
  const [mounted, setMounted] = useState(false);

  useEffect(() => {
    // Check user's browser theme preference
    const prefersDark = window.matchMedia(
      "(prefers-color-scheme: dark)",
    ).matches;
    const initialTheme = prefersDark ? "dark" : "light";

    setTheme(initialTheme);
    document.documentElement.classList.toggle("dark", initialTheme === "dark");

    setMounted(true);
  }, []);

  useEffect(() => {
    if (mounted) {
      // Save theme preference to localStorage
      localStorage.setItem("theme", theme);
      // Toggle dark class on HTML element
      document.documentElement.classList.toggle("dark", theme === "dark");
    }
  }, [theme, mounted]);

  const toggleTheme = () => {
    setTheme((prev) => (prev === "light" ? "dark" : "light"));
  };

  return (
    <ThemeContext.Provider value={{ theme, toggleTheme, mounted }}>
      {children}
    </ThemeContext.Provider>
  );
}

export function useTheme() {
  const context = useContext(ThemeContext);
  if (context === undefined) {
    throw new Error("useTheme must be used within a ThemeProvider");
  }
  return context;
}
```

**Key Points:**
- The component uses `"use client"` directive since it uses browser APIs
- It checks the user's system theme preference on initial load
- It saves the theme preference to localStorage
- It toggles the `dark` class on the `<html>` element
- The `mounted` state prevents hydration mismatches

## Step 3: Wrap Your App with ThemeProvider

In your `app/layout.tsx`, wrap your app with the ThemeProvider and add `suppressHydrationWarning` to prevent browser extension hydration errors:

```tsx
import { ThemeProvider } from "@/components/ThemeProvider";
import type { Metadata } from "next";
import { Geist, Geist_Mono } from "next/font/google";
import "./globals.css";

const geistSans = Geist({
  variable: "--font-geist-sans",
  subsets: ["latin"],
});

const geistMono = Geist_Mono({
  variable: "--font-geist-mono",
  subsets: ["latin"],
});

export const metadata: Metadata = {
  title: "Create Next App",
  description: "Generated by create next app",
};

export default function RootLayout({
  children,
}: Readonly<{
  children: React.ReactNode;
}>) {
  return (
    <html
      lang="en"
      className={`${geistSans.variable} ${geistMono.variable} h-full antialiased`}
    >
      <body className="min-h-full flex flex-col" suppressHydrationWarning>
        <ThemeProvider>{children}</ThemeProvider>
      </body>
    </html>
  );
}
```

**Important:** The `suppressHydrationWarning` prop on the `<body>` tag prevents hydration errors caused by browser extensions that modify the HTML.

## Step 4: Use Dark Mode Classes in Your Components

Use Tailwind's `dark:` prefix to apply dark mode styles:

```tsx
export default function MyComponent() {
  return (
    <div className="min-h-screen bg-zinc-50 dark:bg-black font-sans">
      <h1 className="text-zinc-900 dark:text-zinc-100">
        This text changes color in dark mode
      </h1>
      <p className="text-zinc-600 dark:text-zinc-400">
        This paragraph also changes color
      </p>
      <button className="bg-white dark:bg-zinc-900 text-zinc-900 dark:text-zinc-100">
        Button with dark mode styles
      </button>
    </div>
  );
}
```

## Step 5: Add Theme Toggle Button

Create a theme toggle button using the `useTheme` hook:

```tsx
"use client";

import { useTheme } from "@/components/ThemeProvider";

export default function ThemeToggle() {
  const { theme, toggleTheme, mounted } = useTheme();

  return (
    <button
      onClick={toggleTheme}
      className="p-2 rounded-lg border border-zinc-200 dark:border-zinc-800 hover:bg-zinc-100 dark:hover:bg-zinc-800 transition-colors"
      aria-label="Toggle theme"
    >
      {!mounted ? (
        <div className="w-5 h-5" />
      ) : theme === "dark" ? (
        // Sun icon (shown in dark mode to switch to light)
        <svg
          className="w-5 h-5 text-zinc-900 dark:text-zinc-100"
          fill="none"
          stroke="currentColor"
          viewBox="0 0 24 24"
        >
          <path
            strokeLinecap="round"
            strokeLinejoin="round"
            strokeWidth={2}
            d="M12 3v1m0 16v1m9-9h-1M4 12H3m15.364 6.364l-.707-.707M6.343 6.343l-.707-.707m12.728 0l-.707.707M6.343 17.657l-.707.707M16 12a4 4 0 11-8 0 4 4 0 018 0z"
          />
        </svg>
      ) : (
        // Moon icon (shown in light mode to switch to dark)
        <svg
          className="w-5 h-5 text-zinc-900 dark:text-zinc-100"
          fill="none"
          stroke="currentColor"
          viewBox="0 0 24 24"
        >
          <path
            strokeLinecap="round"
            strokeLinejoin="round"
            strokeWidth={2}
            d="M20.354 15.354A9 9 0 018.646 3.646 9.003 9.003 0 0012 21a9.003 9.003 0 008.354-5.646z"
          />
        </svg>
      )}
    </button>
  );
}
```

**Icon Logic:**
- Sun icon shows when in dark mode (clicking switches to light)
- Moon icon shows when in light mode (clicking switches to dark)
- This follows the pattern of showing what you'll get, not what you have

## Step 6: Access Theme State in Any Component

Use the `useTheme` hook to access the theme state and toggle function in any component:

```tsx
"use client";

import { useTheme } from "@/components/ThemeProvider";

export default function MyComponent() {
  const { theme, toggleTheme } = useTheme();

  return (
    <div>
      <p>Current theme: {theme}</p>
      <button onClick={toggleTheme}>Toggle Theme</button>
    </div>
  );
}
```

## How It Works

1. **Initialization:** On app load, the ThemeProvider checks the user's browser theme preference using `window.matchMedia("(prefers-color-scheme: dark)")` and sets the initial theme accordingly.

2. **Theme Toggling:** When the toggle button is clicked, the `toggleTheme` function switches between "light" and "dark" themes.

3. **DOM Update:** The theme change triggers a useEffect that:
   - Updates the `dark` class on the `<html>` element
   - Saves the preference to localStorage

4. **Style Application:** Tailwind CSS v4's `@custom-variant` directive ensures that all `dark:` prefixed classes are applied when the `.dark` class is present on the HTML element.

5. **CSS Variables:** The CSS variables (`--background`, `--foreground`) change based on the `:root.dark` selector, allowing for dynamic theming.

## Customization

### Change Default Colors

Modify the CSS variables in `globals.css`:

```css
:root {
  --background: #ffffff;  /* Light mode background */
  --foreground: #171717;  /* Light mode text */
}

:root.dark {
  --background: #0a0a0a;  /* Dark mode background */
  --foreground: #ededed;  /* Dark mode text */
}
```

### Persist User Choice Across Visits

The current implementation saves to localStorage. If you want to use the saved preference instead of always defaulting to browser preference, modify the ThemeProvider:

```tsx
useEffect(() => {
  const savedTheme = localStorage.getItem("theme") as Theme | null;
  const prefersDark = window.matchMedia(
    "(prefers-color-scheme: dark)",
  ).matches;
  // Use saved theme if available, otherwise use system preference
  const initialTheme = savedTheme || (prefersDark ? "dark" : "light");

  setTheme(initialTheme);
  document.documentElement.classList.toggle("dark", initialTheme === "dark");

  setMounted(true);
}, []);
```

### Add System Theme Detection on Change

To detect when the user changes their system theme while the app is running:

```tsx
useEffect(() => {
  const mediaQuery = window.matchMedia("(prefers-color-scheme: dark)");
  const handleChange = (e: MediaQueryListEvent) => {
    if (!localStorage.getItem("theme")) {
      setTheme(e.matches ? "dark" : "light");
    }
  };

  mediaQuery.addEventListener("change", handleChange);
  return () => mediaQuery.removeEventListener("change", handleChange);
}, []);
```

## Troubleshooting

### Dark Mode Not Working

- Ensure `@custom-variant dark (&:where(.dark, .dark *));` is added to your globals.css
- Check that the `<html>` element has the `dark` class when in dark mode (use browser DevTools)
- Verify that Tailwind CSS v4 is properly installed

### Hydration Errors

- Add `suppressHydrationWarning` to the `<body>` tag in layout.tsx
- This is usually caused by browser extensions modifying the HTML

### Flash of Wrong Theme on Load

- The `mounted` state in ThemeProvider prevents rendering theme-dependent content until the theme is initialized
- Ensure your components check `mounted` before rendering theme-specific content
